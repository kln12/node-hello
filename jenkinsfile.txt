            }
        }
        
        stage('Build') {
            steps {

            }
        }
        
        stage('Test') {
            steps {
                sh 'npm test'
            }
        }
        
        stage('Deploy') {
            steps {
                // Add your deployment steps here

                // sh 'ssh user@server "cd /path/to/app && git pull && npm install && npm run build"'
            }
        }
    }
    
    post {
        success {
            echo 'Build successful! Deploying...'
            // Trigger deployment here if needed
        }

        }
    }
